01. Select Box

<div class="selectBox" data-form="selectBox"> // class="selectBox" / data-form="selectBox" 필요 <select title="검색어 분류" id="select0"> <option>전체</option> <option>제목</option> <option>내용</option> </select> </div>

02. Select Box Index 확인

<div class="selectBox" data-form="selectBox"> <select title="검색어 분류" id="select1"> <option>전체</option> <option>제목</option> <option>내용</option> </select> </div> <script> $("#select1").change(function(){ var idx = $(this).find("option:selected").index(); alert("select Index: " + idx); }); </script>

03. Select Box 동적생성

<button class="btn btnBlue" id="add">select박스 동적 생성</button> <div class="grid"></div> <!-- 동적으로 생성된 select박스 추가할 위치 --> <script> $("#add").click(function(){ var html = ""; html += "<div class='selectBox' data-form='selectBox'>" html += "<select id='select2'>" // 동적 생성 후 사용 할 id="select2" html += " <option>추가1</option>"; html += " <option>추가2</option>"; html += " <option>추가3</option>"; html += "</select>" html += "</div>" $(".grid").append(html); $("#select2").update(); // 동적 생성 후 select박스 update }) </script>

04. Select Box 옵션변경

<div class="selectBox" data-form="selectBox"> <select id="select3"> <option>서울</option> <option>경기도</option> <option>강원도</option> <option>경상북도</option> </select> </div> <div class="selectBox" data-form="selectBox"> <select id="select4"> <option>강남구</option> <option>서대문구</option> <option>마포구</option> <option>동대문구</option> <option>관악구</option> </select> </div> <script> $(document).on("change", "#select3", function(){ var idx = $(this).find("option:selected").index(); $("#select4").find("option").remove(); if(idx == 0){ var html = ""; html += "<option>강남구</option>" html += "<option>서대문구</option>" html += "<option>마포구</option>" $("#select4").append(html); } if(idx == 1){ var html = ""; html += "<option>일산</option>" html += "<option>남양주</option>" html += "<option>가평</option>" $("#select4").append(html); } $("#select4").update(); }); </script>

05. Select Box 비활성화
